```python
gwc = get_word_combinations(headless=True) # headless=False可見爬蟲的視窗

m1='' # 名的第一個字須包含的字義, ''表示不限制
m2='勤' # 名的第二個字須包含的字義, ''表示不限制
head1='' # 名的第一個字的部首, ''表示不限制
head2='心' # 名的第二個字的部首

# 取得結果: (default) 含評分; do_scoring=True
# tops表示取前幾筆，如果(default)tops=None不限制數量。 !!注意!! 限制數量是未依分數排序前就限制了。
res = gwc.go(m1, m2, head1, head2, do_scoring=True, tops=10)
```
